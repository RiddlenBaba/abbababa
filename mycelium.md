document.addEventListener('DOMContentLoaded', function() {
    const demoButtons = document.querySelectorAll('.demo-scenario');
    const demoContents = document.querySelectorAll('.demo-content');

    demoButtons.forEach(button => {
        button.addEventListener('click', function() {
            // Remove active class from all buttons and content
            demoButtons.forEach(btn => btn.classList.remove('active'));
            demoContents.forEach(content => content.classList.remove('active'));

            // Add active class to clicked button
            this.classList.add('active');

            // Show corresponding content
            const scenario = this.dataset.scenario;
            document.getElementById(scenario).classList.add('active');
        });
    });

    // Network animation
    function animateNetwork() {
        const connections = document.querySelectorAll('.network-connection');
        connections.forEach((conn, index) => {
            setTimeout(() => {
                conn.style.opacity = '1';
                conn.style.animation = 'pulse 2s infinite';
            }, index * 500);
        });
    }

    // Start animation when page loads
    setTimeout(animateNetwork, 1000);
});
